Exercise 7 - Chess Board using Nested Loop

<html>

<head>
    <title>Chessboard</title>
</head>

<body>

<table border="1" cellspacing="0" cellpadding="20">

<?php

for($row = 1; $row <= 8; $row++)
{
    echo "<tr>";

    for($col = 1; $col <= 8; $col++)
    {
        $total = $row + $col;

        if($total % 2 == 0)
            echo "<td bgcolor='white'></td>";
        else
            echo "<td bgcolor='black'></td>";
    }

    echo "</tr>";
}

?>

</table>

</body>
</html>

<?php

if(isset($_POST['date1']) && isset($_POST['date2']))
{
    $date1 = new DateTime($_POST['date1']);
    $date2 = new DateTime($_POST['date2']);

    $diff = $date1->diff($date2);

    echo "Years : " . $diff->y . "<br>";
    echo "Months : " . $diff->m . "<br>";
    echo "Days : " . $diff->d . "<br>";
    echo "Total Days : " . $diff->days;
}
